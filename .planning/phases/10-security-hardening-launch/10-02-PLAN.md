  ```javascript
  /** @type {import('next').NextConfig} */

  const isProd = process.env.NODE_ENV === "production";

  // Derive the backend API origin for CSP connect-src from NEXT_PUBLIC_API_URL.
  // Falls back to a localhost default for dev. In prod this MUST be set
  // (e.g. https://api.famelogistics.com/api -> https://api.famelogistics.com).
  const apiUrl = process.env.NEXT_PUBLIC_API_URL || "http://localhost:8000/api";
  let apiOrigin = "http://localhost:8000";
  try {
    apiOrigin = new URL(apiUrl).origin;
  } catch (_e) {
    // leave default
  }

  // Content Security Policy
  // - Dev: permissive — Next.js HMR needs 'unsafe-eval' and ws: for websockets
  // - Prod: locks down to self + the API origin; allows Google Fonts; no eval
  const cspDirectives = isProd
    ? [
        "default-src 'self'",
        "base-uri 'self'",
        "form-action 'self'",
        "frame-ancestors 'none'",
        // Next.js inlines a tiny bootstrap script; 'unsafe-inline' kept minimal
        "script-src 'self' 'unsafe-inline'",
        "style-src 'self' 'unsafe-inline' https://fonts.googleapis.com",
        "font-src 'self' https://fonts.gstatic.com data:",
        "img-src 'self' data: blob: https:",
        `connect-src 'self' ${apiOrigin}`,
        "object-src 'none'",
        "upgrade-insecure-requests",
      ].join("; ")
    : [
        "default-src 'self'",
        "script-src 'self' 'unsafe-inline' 'unsafe-eval'",
        "style-src 'self' 'unsafe-inline' https://fonts.googleapis.com",
        "font-src 'self' https://fonts.gstatic.com data:",
        "img-src 'self' data: blob: https:",
        "connect-src 'self' http://localhost:* ws://localhost:* https:",
        "frame-ancestors 'none'",
      ].join("; ");

  const securityHeaders = [
    { key: "Content-Security-Policy", value: cspDirectives },
    { key: "X-Content-Type-Options", value: "nosniff" },
    { key: "X-Frame-Options", value: "DENY" },
    { key: "Referrer-Policy", value: "strict-origin-when-cross-origin" },
    {
      key: "Permissions-Policy",
      value: "camera=(), microphone=(), geolocation=(), interest-cohort=()",
    },
  ];

  if (isProd) {
    securityHeaders.push({
      key: "Strict-Transport-Security",
      value: "max-age=31536000; includeSubDomains; preload",
    });
  }

  /** @type {import('next').NextConfig} */
  const nextConfig = {
    output: "standalone",
    poweredByHeader: false,
    async headers() {
      return [
        {
          source: "/:path*",
          headers: securityHeaders,
        },
      ];
    },
  };

  module.exports = nextConfig;
  ```

  Notes:
  - Do NOT remove `output: "standalone"` — the Docker build depends on it.
  - Do NOT switch this file to TypeScript or ESM; Next 14.2 with CommonJS config is what the project uses.
  - Do NOT add `poweredByHeader: true`. It MUST be false.
  - If the developer later moves `NEXT_PUBLIC_API_URL` onto Vercel it will be read at build time; the config pulls the origin out with `new URL(...).origin`, which tolerates both absolute URLs and trailing paths.
