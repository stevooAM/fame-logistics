  <action>
    Create `backend/reports/views.py` with three APIViews. Each view:
    1. Requires `IsAnyRole` permission (same as Phase 7 read-only export actions)
    2. Parses `date_from` and `date_to` from query params (ISO date strings `YYYY-MM-DD`)
    3. Defaults to first and last day of current calendar month when params absent
    4. Returns 400 if dates are provided but malformed
    5. Returns 400 if `date_from > date_to`
    6. Calls the appropriate query function from `reports.queries`
    7. Returns JSON via `Response()`

    ```python
    """
    Report API views for Fame FMS.

    Endpoints:
      GET /api/reports/customer-activity/   RPT-01
      GET /api/reports/job-status/           RPT-02
      GET /api/reports/revenue/              RPT-03

    All endpoints accept:
      date_from=YYYY-MM-DD  (default: first day of current month)
      date_to=YYYY-MM-DD    (default: last day of current month)

    Revenue endpoint also accepts:
      currency_code=GHS|USD|...  (default: all currencies combined)

    Customer activity endpoint also accepts:
      customer_id=<int>  (default: all customers)

    Permissions: IsAnyRole (Admin, Finance, Operations)
    """

    from datetime import date, datetime
    from calendar import monthrange

    from rest_framework.response import Response
    from rest_framework.views import APIView

    from core.permissions import IsAnyRole

    from .queries import customer_activity_query, job_status_query, revenue_query


    def _parse_dates(request) -> tuple[date, date] | tuple[None, None, Response]:
        """
        Parse date_from and date_to from query params.
        Returns (date_from, date_to) or raises ValueError-equivalent via early return.

        Defaults to first/last day of current month when params absent.
        Returns (date_from, date_to, error_response) tuple on parse failure
        — caller checks `len(result) == 3` to detect error path.
        """
        today = date.today()
        first_of_month = today.replace(day=1)
        _, last_day = monthrange(today.year, today.month)
        last_of_month = today.replace(day=last_day)

        raw_from = request.query_params.get("date_from")
        raw_to = request.query_params.get("date_to")

        try:
            date_from = datetime.strptime(raw_from, "%Y-%m-%d").date() if raw_from else first_of_month
            date_to = datetime.strptime(raw_to, "%Y-%m-%d").date() if raw_to else last_of_month
        except ValueError:
            return None, None, Response(
                {"error": "Invalid date format. Use YYYY-MM-DD."},
                status=400,
            )

        if date_from > date_to:
            return None, None, Response(
                {"error": "date_from must be on or before date_to."},
                status=400,
            )

        return date_from, date_to


    class CustomerActivityView(APIView):
        """
        GET /api/reports/customer-activity/

        Query params:
          date_from, date_to  — ISO date, defaults to current month
          customer_id         — optional int, scopes to single customer

        Response: list of customer activity rows
        """
        permission_classes = [IsAnyRole]

        def get(self, request):
            result = _parse_dates(request)
            if len(result) == 3:
                return result[2]
            date_from, date_to = result

            customer_id = request.query_params.get("customer_id")
            if customer_id:
                try:
                    customer_id = int(customer_id)
                except (ValueError, TypeError):
                    return Response({"error": "customer_id must be an integer."}, status=400)

            rows = customer_activity_query(date_from, date_to, customer_id=customer_id)
            return Response(
                {
                    "date_from": date_from.isoformat(),
                    "date_to": date_to.isoformat(),
                    "rows": rows,
                    "count": len(rows),
                }
            )


    class JobStatusView(APIView):
        """
        GET /api/reports/job-status/

        Query params:
          date_from, date_to  — ISO date, defaults to current month

        Response: list of (status, job_type) group rows
        Always shows all status/type groups — no per-status filter (user decision).
        """
        permission_classes = [IsAnyRole]

        def get(self, request):
            result = _parse_dates(request)
            if len(result) == 3:
                return result[2]
            date_from, date_to = result

            rows = job_status_query(date_from, date_to)
            return Response(
                {
                    "date_from": date_from.isoformat(),
                    "date_to": date_to.isoformat(),
                    "rows": rows,
                }
            )


    class RevenueView(APIView):
        """
        GET /api/reports/revenue/

        Query params:
          date_from, date_to  — ISO date, defaults to current month
          currency_code       — e.g. GHS, USD; omit for all currencies combined

        Response: {period_rows, period_totals, customer_rows, customer_totals, date_from, date_to, currency_code}
        Two-section structure (period summary + per-customer breakdown) mirrors Accounts summary page.
        """
        permission_classes = [IsAnyRole]

        def get(self, request):
            result = _parse_dates(request)
            if len(result) == 3:
                return result[2]
            date_from, date_to = result

            currency_code = request.query_params.get("currency_code") or None

            data = revenue_query(date_from, date_to, currency_code=currency_code)
            return Response(
                {
                    "date_from": date_from.isoformat(),
                    "date_to": date_to.isoformat(),
                    "currency_code": currency_code,
                    **data,
                }
            )
    ```

    Important:
    - `_parse_dates` returns a 2-tuple on success and 3-tuple on error. The caller checks `len(result) == 3` to detect the error path — this is a simple, no-exception pattern.
    - `IsAnyRole` is imported from `core.permissions` — same class used throughout Phase 7.
    - No serializer classes needed; the query functions already return plain dicts.
    - `currency_code` is passed as `None` (not empty string) to `revenue_query` so the query function can use `if currency_code:` to conditionally apply the filter.
