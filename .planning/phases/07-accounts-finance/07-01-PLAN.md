```python
from django.db import transaction
from django.utils import timezone

def generate_invoice_number() -> str:
    """
    Generate the next invoice number in INV-{YEAR}-{SEQUENCE:05d} format.
    Uses select_for_update() to avoid race conditions on concurrent writes.
    Caller is responsible for wrapping in an atomic block when critical.
    """
    year = timezone.now().year
    prefix = f"INV-{year}-"
    latest = (
        Invoice.objects.select_for_update()
        .filter(invoice_number__startswith=prefix)
        .order_by("-invoice_number")
        .first()
    )
    if latest is None:
        sequence = 1
    else:
        try:
            sequence = int(latest.invoice_number.split("-")[-1]) + 1
        except (ValueError, IndexError):
            sequence = 1
    return f"{prefix}{sequence:05d}"
```

Place this function ABOVE the `Invoice` class so the class can reference it (forward reference works because the function body is only evaluated when called).

**3. Add three balance helper methods to `Invoice`**:

```python
from decimal import Decimal
from django.db.models import Sum

def invoiced_total(self) -> Decimal:
    """Return this invoice's amount as a Decimal (always non-null)."""
    return self.amount or Decimal("0")

def paid_total(self) -> Decimal:
    """Return Decimal sum of all related Payment.amount values, 0 if none."""
    total = self.payments.aggregate(total=Sum("amount"))["total"]
    return total or Decimal("0")

def balance(self) -> Decimal:
    """Return invoiced_total() - paid_total() as a Decimal."""
    return self.invoiced_total() - self.paid_total()
```

**4. Add an `InvoiceQuerySet` / custom Manager method `outstanding_for_customer(customer_id)`** on Invoice:

```python
class InvoiceQuerySet(models.QuerySet):
    def outstanding_for_customer(self, customer_id: int) -> dict:
        """
        Aggregate invoiced/paid/balance Decimals for one customer across all
        their non-cancelled invoices.

        Returns: {"invoiced": Decimal, "paid": Decimal, "balance": Decimal}
        """
        from decimal import Decimal as _D
        qs = self.filter(customer_id=customer_id).exclude(status=Invoice.CANCELLED)
        invoiced = qs.aggregate(total=Sum("amount"))["total"] or _D("0")
        paid = (
            Payment.objects.filter(invoice__customer_id=customer_id)
            .exclude(invoice__status=Invoice.CANCELLED)
            .aggregate(total=Sum("amount"))["total"]
            or _D("0")
        )
        return {"invoiced": invoiced, "paid": paid, "balance": invoiced - paid}
```

Wire the manager onto Invoice: `objects = InvoiceQuerySet.as_manager()`.
