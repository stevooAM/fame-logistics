<objective>
Build the Outstanding Balances UI for Finance staff: a per-customer balances table (with server-side pagination, search, sort, red/amber styling for non-zero balances, totals footer, per-customer currency code) plus a drill-down page showing one customer's full invoice list with their balance. **LOCKED API CONTRACTS (checker-fix iteration 1) — all types below MUST match these exactly:**
- `/api/accounts/balances/` row shape: `{customer_id, customer_name, invoiced_total, paid_total, balance, invoice_count, currency_code}`. Field is `currency_code` (string), populated from `customer.currency_preference.code` defaulting to `"GHS"`.
- `/api/accounts/balances/{customer_id}/` shape: `{customer: {id, company_name, email, phone_number, tin}, invoiced_total, paid_total, balance, invoice_count, invoices: [...]}`. - Drill-down invoice rows are NON-interactive in this plan. The `?invoice={id}` cross-page link-back is DEFERRED to Phase 8 (see Task 2 note).

Output: New `balances/` subtree under `accounts/` in the frontend; typed API client; AG Grid table with drill-down routing.
</objective>

**1. Create `frontend/src/lib/balances-api.ts`** — typed client. **All shapes MUST align with the locked 07-03 backend contract (checker-fix iteration 1).** Field names below are authoritative — do not rename or invent:

```typescript
import { apiFetch, apiFetchBlob } from "@/lib/api";

/**
 * One row in /api/accounts/balances/ response.
 * Matches 07-03 `_build_balance_rows` row dict exactly.
 */
export interface CustomerBalance {
  customer_id: number;
  customer_name: string;
  invoice_count: number;
  invoiced_total: string;  // Decimal as string from DRF
  paid_total: string;
  balance: string;
  currency_code: string;   // from customer.currency_preference.code, default "GHS"
}

export interface CustomerBalancesResponse {
  count: number;
  next: string | null;
  previous: string | null;
  results: CustomerBalance[];
}

/**
 * One invoice row inside /api/accounts/balances/{customer_id}/ response.
 * Produced by backend InvoiceListSerializer (from 07-02). Field names are
 * locked: paid_total + balance + amount + currency_code (NOT paid/amount_paid).
 */
export interface BalanceInvoice {
  id: number;
  invoice_number: string;
  job: number;
  job_number: string | null;
  customer: number;
  customer_name: string | null;
  amount: string;
  paid_total: string;
  balance: string;
  currency: number | null;
  currency_code: string | null;
  status: string;
  issue_date: string;
  due_date: string;
  created_at: string;
}

/**
 * Full shape of /api/accounts/balances/{customer_id}/ — matches 07-03
 * `balance_detail()` action exactly (checker-fix iteration 1 lock).
 * Note: customer has email/phone_number/tin (snake_case, matches backend
 * Customer model). No email/phone/tin on individual invoices.
 */
export interface CustomerBalanceDetailResponse {
  customer: {
    id: number;
    company_name: string;
    email: string | null;
    phone_number: string | null;
    tin: string | null;
  };
  invoiced_total: string;
  paid_total: string;
  balance: string;
  invoice_count: number;
  invoices: BalanceInvoice[];
}

export interface BalancesFilters {
  search?: string;
  ordering?: "balance" | "-balance" | "invoiced_total" | "-invoiced_total" | "paid_total" | "-paid_total" | "customer_name" | "-customer_name";
  page?: number;
  page_size?: number;
  include_zero?: boolean;
}

export async function fetchBalances(
  filters: BalancesFilters = {},
  signal?: AbortSignal,
): Promise<CustomerBalancesResponse> {
  const params = new URLSearchParams();
  if (filters.search) params.set("search", filters.search);
  if (filters.ordering) params.set("ordering", filters.ordering);
  if (filters.page) params.set("page", String(filters.page));
  if (filters.page_size) params.set("page_size", String(filters.page_size));
  if (filters.include_zero) params.set("include_zero", "true");
  const qs = params.toString();
  return apiFetch<CustomerBalancesResponse>(
    `/api/accounts/balances/${qs ? `?${qs}` : ""}`,
    { signal },
  );
}

export async function fetchBalanceDetail(
  customerId: number,
  signal?: AbortSignal,
): Promise<CustomerBalanceDetailResponse> {
  return apiFetch<CustomerBalanceDetailResponse>(
    `/api/accounts/balances/${customerId}/`,
    { signal },
  );
}

export async function exportBalancesBlob(
  filters: BalancesFilters = {},
  format: "xlsx" | "csv" = "xlsx",
): Promise<Blob> {
  const params = new URLSearchParams();
  params.set("format", format);
  if (filters.search) params.set("search", filters.search);
  if (filters.ordering) params.set("ordering", filters.ordering);
  return apiFetchBlob(`/api/accounts/balances/export/?${params.toString()}`);
}
```
