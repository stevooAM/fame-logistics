class InvoiceListSerializer(serializers.ModelSerializer):
    """Lightweight payload for the invoice list grid."""
    customer_name = serializers.SerializerMethodField()
    job_number = serializers.SerializerMethodField()
    balance = serializers.SerializerMethodField()
    paid_total = serializers.SerializerMethodField()
    currency_code = serializers.SerializerMethodField()

    class Meta:
        model = Invoice
        fields = [
            "id", "invoice_number", "customer", "customer_name",
            "job", "job_number", "amount", "paid_total", "balance",
            "currency", "currency_code", "status", "issue_date", "due_date",
            "created_at",
        ]
        read_only_fields = fields

    def get_customer_name(self, obj):
        return obj.customer.company_name if obj.customer_id else None

    def get_job_number(self, obj):
        return obj.job.job_number if obj.job_id else None

    def get_balance(self, obj):
        return str(obj.balance())

    def get_paid_total(self, obj):
        return str(obj.paid_total())

    def get_currency_code(self, obj):
        return getattr(obj.currency, "code", None)


class PaymentSerializer(serializers.ModelSerializer):
    recorded_by_name = serializers.SerializerMethodField(read_only=True)

    class Meta:
        model = Payment
        fields = [
            "id", "invoice", "amount", "payment_date", "payment_method",
            "reference", "notes", "recorded_by", "recorded_by_name",
            "created_at",
        ]
        read_only_fields = ["id", "recorded_by", "recorded_by_name", "created_at"]

    def get_recorded_by_name(self, obj):
        if obj.recorded_by_id and obj.recorded_by:
            full = obj.recorded_by.get_full_name()
            return full if full else obj.recorded_by.username
        return None

    def validate_amount(self, value):
        if value is None or value <= 0:
            raise serializers.ValidationError("Amount must be greater than zero.")
        return value

    def validate(self, attrs):
        invoice = attrs.get("invoice")
        amount = attrs.get("amount")
        if invoice and amount:
            outstanding = invoice.balance()
            if amount > outstanding:
                raise serializers.ValidationError({
                    "amount": (
                        f"Amount {amount} exceeds outstanding balance {outstanding}."
                    )
                })
        return attrs


class InvoiceSerializer(serializers.ModelSerializer):
    """Full payload for invoice detail / update operations."""
    customer_name = serializers.SerializerMethodField(read_only=True)
    job_number = serializers.SerializerMethodField(read_only=True)
    balance = serializers.SerializerMethodField(read_only=True)
    paid_total = serializers.SerializerMethodField(read_only=True)
    payments = PaymentSerializer(many=True, read_only=True)
    currency_code = serializers.SerializerMethodField(read_only=True)

    class Meta:
        model = Invoice
        fields = [
            "id", "invoice_number", "job", "job_number", "customer", "customer_name",
            "amount", "currency", "currency_code", "status", "issue_date", "due_date",
            "notes", "paid_total", "balance", "payments",
            "created_by", "created_at", "updated_at",
        ]
        read_only_fields = [
            "id", "invoice_number", "customer", "job", "created_by",
            "created_at", "updated_at",
        ]

    # mirrors InvoiceListSerializer methods (copy them)
    # … get_customer_name / get_job_number / get_balance / get_paid_total / get_currency_code

    def to_representation(self, instance):
        data = super().to_representation(instance)
        data["customer_detail"] = (
            {"id": instance.customer.id, "company_name": instance.customer.company_name}
            if instance.customer_id else None
        )
        data["job_detail"] = (
            {"id": instance.job.id, "job_number": instance.job.job_number,
             "status": instance.job.status}
            if instance.job_id else None
        )
        return data

    issue_date = serializers.DateField(required=False)
    due_date = serializers.DateField()

```python
class InvoiceViewSet(AuditLogMixin, ModelViewSet):
    pagination_class = AccountsPagination

    def get_permissions(self):
        # Read = any role (Operations needs to see invoice number on job details).
        # Write = Admin or Finance only.
        if self.action in {"list", "retrieve"}:
            return [IsAnyRole()]
        return [IsAdminOrFinance()]

    def get_serializer_class(self):
        if self.action == "list":
            return InvoiceListSerializer
        return InvoiceSerializer

    def get_queryset(self):
        qs = (
            Invoice.objects.select_related("customer", "job", "currency", "created_by")
            .prefetch_related("payments")
            .all()
            .order_by("-created_at")
        )
        p = self.request.query_params

        search = p.get("search", "").strip()
        if search:
            qs = qs.filter(
                Q(invoice_number__icontains=search)
                | Q(customer__company_name__icontains=search)
                | Q(job__job_number__icontains=search)
            )

        for key, field in (("status", "status"), ("customer_id", "customer_id"),
                           ("job_id", "job_id")):
            val = p.get(key, "").strip()
            if val:
                if field.endswith("_id"):
                    try:
                        qs = qs.filter(**{field: int(val)})
                    except (TypeError, ValueError):
                        pass
                else:
                    qs = qs.filter(**{field: val})

        date_from = p.get("date_from", "").strip()
        if date_from:
            qs = qs.filter(issue_date__gte=date_from)
        date_to = p.get("date_to", "").strip()
        if date_to:
            qs = qs.filter(issue_date__lte=date_to)

        ordering = p.get("ordering", "").strip()
        if ordering in {
            "invoice_number", "-invoice_number", "issue_date", "-issue_date",
            "amount", "-amount", "created_at", "-created_at", "status", "-status",
        }:
            qs = qs.order_by(ordering)

        return qs

    def destroy(self, request, *args, **kwargs):
        return Response(
            {"detail": "Invoice deletion is not permitted. Use status CANCELLED."},
            status=status.HTTP_405_METHOD_NOT_ALLOWED,
        )

    def perform_create(self, serializer):
        # Bypass mixin double-log; serializer.save() with created_by
        serializer.save(created_by=self.request.user)
        self._log_action(self.request, "CREATE", serializer.instance)

    @action(detail=False, methods=["post"], url_path="generate")
    def generate(self, request):
        """
        POST /api/accounts/invoices/generate/
        Body: {job_id, amount, currency_id?, issue_date?, due_date, notes?}
        Verifies the job has an APPROVED ApprovalQueue entry. Auto-generates
        invoice_number inside transaction.atomic so select_for_update is honoured.
        """
        ser = GenerateInvoiceSerializer(data=request.data)
        ser.is_valid(raise_exception=True)
        job_id = ser.validated_data["job_id"]

        try:
            job = Job.objects.select_related("customer").get(pk=job_id)
        except Job.DoesNotExist:
            return Response({"detail": "Job not found."},
                            status=status.HTTP_404_NOT_FOUND)

        approved = ApprovalQueue.objects.filter(
            job_id=job_id, status=ApprovalQueue.APPROVED,
        ).exists()
        if not approved:
            return Response(
                {"detail": "Job has no approved approval entry. "
                           "Only approved jobs can be invoiced."},
                status=status.HTTP_400_BAD_REQUEST,
            )

        with transaction.atomic():
            invoice = Invoice.objects.create(
                job=job,
                customer=job.customer,
                amount=ser.validated_data["amount"],
                currency_id=ser.validated_data.get("currency_id"),
                issue_date=ser.validated_data.get("issue_date") or _date.today(),
                due_date=ser.validated_data["due_date"],
                notes=ser.validated_data.get("notes", ""),
                status=Invoice.DRAFT,
                created_by=request.user,
            )

        self._log_action(request, "CREATE", invoice,
                         description=f"Invoice generated for job {job.job_number}")

        return Response(InvoiceSerializer(invoice, context={"request": request}).data,
                        status=status.HTTP_201_CREATED)
```

```python
class PaymentViewSet(AuditLogMixin, ModelViewSet):
    serializer_class = PaymentSerializer
    pagination_class = AccountsPagination
    http_method_names = ["get", "post", "head", "options"]

    def get_permissions(self):
        if self.action in {"list", "retrieve"}:
            return [IsAnyRole()]
        return [IsAdminOrFinance()]

    def get_queryset(self):
        qs = (
            Payment.objects.select_related(
                "invoice", "invoice__customer", "recorded_by"
            ).all().order_by("-payment_date", "-created_at")
        )
        p = self.request.query_params
        invoice_id = p.get("invoice_id", "").strip()
        if invoice_id:
            try:
                qs = qs.filter(invoice_id=int(invoice_id))
            except (TypeError, ValueError):
                pass
        customer_id = p.get("customer_id", "").strip()
        if customer_id:
            try:
                qs = qs.filter(invoice__customer_id=int(customer_id))
            except (TypeError, ValueError):
                pass
        return qs

    def perform_create(self, serializer):
        with transaction.atomic():
            payment = serializer.save(recorded_by=self.request.user)
            invoice = payment.invoice
            balance = invoice.balance()
            paid = invoice.paid_total()
            invoiced = invoice.invoiced_total()
            if balance <= Decimal("0"):
                new_status = Invoice.PAID
            elif paid > Decimal("0") and paid < invoiced:
                new_status = Invoice.PARTIAL
            else:
                new_status = invoice.status if invoice.status != Invoice.DRAFT else Invoice.SENT
            if invoice.status != new_status:
                invoice.status = new_status
                invoice.save(update_fields=["status", "updated_at"])
        self._log_action(self.request, "CREATE", payment,
                         description=f"Payment {payment.amount} on {invoice.invoice_number}")
```

**Avoid:**
- Do NOT allow PUT/PATCH/DELETE on Payment (override `http_method_names` as shown).
- Do NOT auto-set `Invoice.status = SENT` on creation in `generate` — leave at DRAFT; status flips happen via payments or explicit user actions.
- Do NOT use `Decimal(amount)` constructed from a float anywhere — DRF DecimalField already returns Decimal.
- Do NOT call `super().perform_create(serializer)` from the AuditLogMixin path — we set `recorded_by` ourselves and call `_log_action` once (mirrors the 05-03 decision in JobViewSet.perform_create).
