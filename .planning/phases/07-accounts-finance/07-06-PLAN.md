**1. Create `frontend/src/lib/summaries-api.ts`** — typed client for the period summary endpoint built in 07-03. **All shapes and param names below are LOCKED to the 07-03 backend contract (checker-fix iteration 1).** Do not rename, do not add fields the backend does not return:

```typescript
import { apiFetch } from "@/lib/api";

export type SummaryPeriod = "month" | "quarter";

/**
 * One row in the wrapped summary response. Field names match the 07-03
 * `summary()` action exactly. `outstanding` is the canonical name (NOT balance).
 * Backend does NOT return invoice_count or payment_count for period rows.
 */
export interface PeriodSummaryRow {
  period_start: string;  // ISO date, first day of month or first day of quarter
  period_end: string;    // ISO date, last day of month/quarter
  period_label: string;  // "Jan 2026" or "Q1 2026"
  invoiced: string;      // Decimal string
  paid: string;
  outstanding: string;   // = invoiced - paid
}

/**
 * Wrapped response from /api/accounts/summary/. Frontend SummaryTable reads
 * `data.totals` directly — backend always returns it, even for empty ranges
 * (totals will be zero strings).
 */
export interface PeriodSummaryResponse {
  period: SummaryPeriod;
  date_from: string;     // echoed back from request, may be empty string
  date_to: string;
  rows: PeriodSummaryRow[];
  totals: {
    invoiced: string;
    paid: string;
    outstanding: string;
  };
}

/**
 * Filters sent to the summary endpoint. Param names MUST be date_from/date_to
 * (NOT start_date/end_date) — backend filters silently ignore other names.
 */
export interface SummaryFilters {
  period: SummaryPeriod;
  date_from?: string;
  date_to?: string;
}

export async function fetchPeriodSummary(
  filters: SummaryFilters,
  signal?: AbortSignal,
): Promise<PeriodSummaryResponse> {
  const params = new URLSearchParams();
  params.set("period", filters.period);
  if (filters.date_from) params.set("date_from", filters.date_from);
  if (filters.date_to) params.set("date_to", filters.date_to);
  return apiFetch<PeriodSummaryResponse>(
    `/api/accounts/summary/?${params.toString()}`,
    { signal },
  );
}
```

**2. Create `SummarySection.tsx`** — client component owning all state and composition. Note: state field names are `date_from`/`date_to` (matches the locked filter shape):

```tsx
"use client";

import { useEffect, useState } from "react";
import type { PeriodSummaryResponse, SummaryPeriod } from "@/lib/summaries-api";
import { fetchPeriodSummary } from "@/lib/summaries-api";
import SummaryToolbar from "./SummaryToolbar";
import SummaryTable from "./SummaryTable";
import SummaryChart from "./SummaryChart";

function defaultRange(period: SummaryPeriod): { date_from: string; date_to: string } {
  const end = new Date();
  const start = new Date();
  if (period === "month") start.setMonth(end.getMonth() - 11);  // trailing 12 months
  else start.setMonth(end.getMonth() - 23);                     // trailing 8 quarters (~24 months)
  start.setDate(1);
  const iso = (d: Date) => d.toISOString().slice(0, 10);
  return { date_from: iso(start), date_to: iso(end) };
}

export default function SummarySection() {
  const [period, setPeriod] = useState<SummaryPeriod>("month");
  const [range, setRange] = useState(() => defaultRange("month"));
  const [data, setData] = useState<PeriodSummaryResponse | null>(null);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState<string | null>(null);

  useEffect(() => {
    const controller = new AbortController();
    setLoading(true);
    setError(null);
    fetchPeriodSummary({ period, ...range }, controller.signal)
      .then((res) => { setData(res); setLoading(false); })
      .catch((err) => {
        if (err.name === "AbortError") return;
        setError(err?.detail || err?.message || "Failed to load summary");
        setLoading(false);
      });
    return () => controller.abort();
  }, [period, range]);

  const handlePeriodChange = (next: SummaryPeriod) => {
    setPeriod(next);
    setRange(defaultRange(next));   // reset range to sensible default for new period
  };

  return (
    <div className="flex flex-col gap-4">
      <SummaryToolbar
        period={period}
        range={range}
        onPeriodChange={handlePeriodChange}
        onRangeChange={setRange}
        loading={loading}
      />
      {error && (
        <div className="rounded border border-red-200 bg-red-50 p-3 text-sm text-red-700">
          {error}
        </div>
      )}
      {data && !error && (
        <>
          <SummaryChart rows={data.rows} />
          <SummaryTable data={data} />
        </>
      )}
      {loading && !data && (
        <div className="text-sm text-[var(--fms-muted)]">Loading summary…</div>
      )}
    </div>
  );
}
```

**4. Create `SummaryTable.tsx`** — simple HTML table:

```tsx
"use client";

import type { PeriodSummaryResponse } from "@/lib/summaries-api";

const fmtMoney = (v: string) =>
  new Intl.NumberFormat("en-GH", {
    style: "currency",
    currency: "GHS",
    minimumFractionDigits: 2,
  }).format(Number(v));

export default function SummaryTable({ data }: { data: PeriodSummaryResponse }) {
  if (data.rows.length === 0) {
    return (
      <div className="rounded border border-[var(--fms-border)] bg-white p-4 text-sm text-[var(--fms-muted)]">
        No invoices in this date range.
      </div>
    );
  }
  return (
    <div className="overflow-x-auto rounded border border-[var(--fms-border)] bg-white">
      <table className="min-w-full text-sm">
        <thead className="bg-[var(--fms-bg)] text-left">
          <tr>
            <th className="px-3 py-2 font-medium">Period</th>
            <th className="px-3 py-2 font-medium text-right">Invoiced (GHS)</th>
            <th className="px-3 py-2 font-medium text-right">Paid (GHS)</th>
            <th className="px-3 py-2 font-medium text-right">Outstanding (GHS)</th>
          </tr>
        </thead>
        <tbody>
          {data.rows.map((r) => (
            <tr key={r.period_start} className="border-t border-[var(--fms-border)]">
              <td className="px-3 py-2">{r.period_label}</td>
              <td className="px-3 py-2 text-right">{fmtMoney(r.invoiced)}</td>
              <td className="px-3 py-2 text-right">{fmtMoney(r.paid)}</td>
              <td className={`px-3 py-2 text-right ${Number(r.outstanding) > 0 ? "text-red-600 font-semibold" : ""}`}>
                {fmtMoney(r.outstanding)}
              </td>
            </tr>
          ))}
        </tbody>
        <tfoot className="border-t-2 border-[var(--fms-border)] bg-[var(--fms-bg)] font-semibold">
          <tr>
            <td className="px-3 py-2">Totals</td>
            <td className="px-3 py-2 text-right">{fmtMoney(data.totals.invoiced)}</td>
            <td className="px-3 py-2 text-right">{fmtMoney(data.totals.paid)}</td>
            <td className="px-3 py-2 text-right">{fmtMoney(data.totals.outstanding)}</td>
          </tr>
        </tfoot>
      </table>
    </div>
  );
}
```

```tsx
import type { PeriodSummaryRow } from "@/lib/summaries-api";

interface Props { rows: PeriodSummaryRow[] }

    ...rows.flatMap((r) => [Number(r.invoiced), Number(r.paid)]),
```
