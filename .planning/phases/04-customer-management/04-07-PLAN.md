```typescript
const handleExport = async (format: "xlsx" | "csv") => {
  const params = new URLSearchParams();
  params.set("format", format);
  if (filters.search) params.set("search", filters.search);
  // ... add other active filters

  const response = await fetch(
    `${API_BASE_URL}/api/customers/export/?${params.toString()}`,
    { credentials: "include" }
  );
  const blob = await response.blob();
  const url = window.URL.createObjectURL(blob);
  const a = document.createElement("a");
  a.href = url;
  a.download = `customers_export.${format === "xlsx" ? "xlsx" : "csv"}`;
  a.click();
  window.URL.revokeObjectURL(url);
};
```

Import API_BASE_URL from @/lib/api. Use raw fetch here (not apiFetch) since we need the blob response, not JSON. Still include credentials.
