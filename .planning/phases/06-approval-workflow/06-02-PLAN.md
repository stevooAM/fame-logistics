    Create backend/approvals/views.py:

    ```python
    from django.utils import timezone
    from rest_framework import status
    from rest_framework.decorators import action
    from rest_framework.response import Response
    from rest_framework.viewsets import GenericViewSet
    from rest_framework.mixins import ListModelMixin, RetrieveModelMixin

    from core.permissions import IsAdminOrOperations, IsAdmin
    from jobs.models import Job, JobAuditTrail, JobStatus
    from .models import ApprovalQueue, ApprovalHistory
    from .serializers import (
        ApprovalQueueSerializer,
        ApprovalHistorySerializer,
        ApprovalActionSerializer,
    )

    class ApprovalViewSet(ListModelMixin, RetrieveModelMixin, GenericViewSet):
        """
        Approval workflow endpoints.

        GET  /api/approvals/                — pending queue (Admin + Operations)
        GET  /api/approvals/{id}/           — single pending item (Admin + Operations)
        POST /api/approvals/{id}/approve/   — approve (Admin + Operations)
        POST /api/approvals/{id}/reject/    — reject with mandatory reason (Admin + Operations)
        GET  /api/approvals/history/        — full history (Admin only)
        GET  /api/approvals/pending-count/  — count for sidebar badge (Admin + Operations)
        """
        serializer_class = ApprovalQueueSerializer
        permission_classes = [IsAdminOrOperations]

        def get_queryset(self):
            return (
                ApprovalQueue.objects
                .filter(status=ApprovalQueue.PENDING)
                .select_related("job", "job__customer", "submitted_by", "submitted_by__profile")
                .order_by("-created_at")
            )

        @action(detail=True, methods=["post"], url_path="approve")
        def approve(self, request, pk=None):
            approval = self.get_object()
            if not approval.is_pending:
                return Response(
                    {"detail": "This approval request has already been actioned."},
                    status=status.HTTP_400_BAD_REQUEST,
                )

            ser = ApprovalActionSerializer(data=request.data)
            ser.is_valid(raise_exception=True)
            comment = ser.validated_data.get("reason", "")

            # Update ApprovalQueue
            approval.status = ApprovalQueue.APPROVED
            approval.reviewed_by = request.user
            approval.reviewed_at = timezone.now()
            approval.save(update_fields=["status", "reviewed_by", "reviewed_at", "updated_at"])

            # Transition Job to IN_PROGRESS
            job = approval.job
            job.status = JobStatus.IN_PROGRESS
            job.save(update_fields=["status", "updated_at"])

            # Record in ApprovalHistory
            ApprovalHistory.objects.create(
                approval=approval,
                action=ApprovalHistory.APPROVED,
                actor=request.user,
                comment=comment,
            )

            # Job audit trail entry
            JobAuditTrail.objects.create(
                job=job,
                action="STATUS_CHANGE",
                actor=request.user,
                description=f"Job approved by {request.user.username}. Status changed to In Progress.",
            )

            return Response(
                ApprovalQueueSerializer(approval).data,
                status=status.HTTP_200_OK,
            )

        @action(detail=True, methods=["post"], url_path="reject")
        def reject(self, request, pk=None):
            approval = self.get_object()
            if not approval.is_pending:
                return Response(
                    {"detail": "This approval request has already been actioned."},
                    status=status.HTTP_400_BAD_REQUEST,
                )

            ser = ApprovalActionSerializer(data=request.data)
            ser.is_valid(raise_exception=True)
            reason = ser.validated_data.get("reason", "").strip()

            if not reason:
                return Response(
                    {"reason": ["A rejection reason is required."]},
                    status=status.HTTP_400_BAD_REQUEST,
                )

            # Update ApprovalQueue
            approval.status = ApprovalQueue.REJECTED
            approval.reviewed_by = request.user
            approval.reviewed_at = timezone.now()
            approval.rejection_reason = reason
            approval.save(update_fields=["status", "reviewed_by", "reviewed_at", "rejection_reason", "updated_at"])

            # Revert Job to DRAFT so originator can edit and resubmit
            job = approval.job
            job.status = JobStatus.DRAFT
            job.save(update_fields=["status", "updated_at"])

            # Record in ApprovalHistory
            ApprovalHistory.objects.create(
                approval=approval,
                action=ApprovalHistory.REJECTED,
                actor=request.user,
                comment=reason,
            )

            # Job audit trail entry
            JobAuditTrail.objects.create(
                job=job,
                action="STATUS_CHANGE",
                actor=request.user,
                description=f"Job rejected by {request.user.username}. Reason: {reason}",
            )

            return Response(
                ApprovalQueueSerializer(approval).data,
                status=status.HTTP_200_OK,
            )

        @action(detail=False, methods=["get"], url_path="history", permission_classes=[IsAdmin])
        def history(self, request):
            """Admin-only full approval history with optional filters."""
            qs = (
                ApprovalHistory.objects
                .select_related("approval__job", "actor", "actor__profile")
                .order_by("-created_at")
            )

            # Optional filter: job_number
            job_number = request.query_params.get("job_number", "").strip()
            if job_number:
                qs = qs.filter(approval__job__job_number__icontains=job_number)

            # Optional date range filter: date_from, date_to (ISO date strings)
            date_from = request.query_params.get("date_from", "")
            date_to = request.query_params.get("date_to", "")
            if date_from:
                try:
                    from datetime import date
                    qs = qs.filter(created_at__date__gte=date.fromisoformat(date_from))
                except ValueError:
                    pass  # Malformed date — silently ignore (consistent with 03-03 pattern)
            if date_to:
                try:
                    from datetime import date, timedelta
                    qs = qs.filter(created_at__lt=date.fromisoformat(date_to) + timedelta(days=1))
                except ValueError:
                    pass

            # Action filter: APPROVED, REJECTED, SUBMITTED
            action_filter = request.query_params.get("action", "").upper()
            if action_filter in [ApprovalHistory.SUBMITTED, ApprovalHistory.APPROVED, ApprovalHistory.REJECTED]:
                qs = qs.filter(action=action_filter)

            serializer = ApprovalHistorySerializer(qs, many=True)
            return Response(serializer.data)

        @action(detail=False, methods=["get"], url_path="pending-count")
        def pending_count(self, request):
            """Returns count of PENDING approvals. Used by sidebar badge polling."""
            count = ApprovalQueue.objects.filter(status=ApprovalQueue.PENDING).count()
            return Response({"count": count})
    ```
