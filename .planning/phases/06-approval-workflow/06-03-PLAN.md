    Create frontend/src/hooks/use-approval-badge.ts:

    ```typescript
    "use client";

    import { useEffect, useState, useRef } from "react";
    import { apiFetch } from "@/lib/api";

    const POLL_INTERVAL_MS = 30_000; // 30 seconds — matches useIdleTimeout activity throttle pattern

    interface PendingCountResponse {
      count: number;
    }

    /**
     * Polls /api/approvals/pending-count/ every 30 seconds.
     * Returns the current count of pending approvals.
     * Safe to call from any component — cleans up interval on unmount.
     *
     * Returns 0 on network failure (non-blocking, badge hides gracefully).
     */
    export function useApprovalBadge(): number {
      const [count, setCount] = useState<number>(0);
      const intervalRef = useRef<ReturnType<typeof setInterval> | null>(null);

      const fetchCount = async () => {
        try {
          const data = await apiFetch<PendingCountResponse>("/api/approvals/pending-count/");
          setCount(data.count ?? 0);
        } catch {
          // Network failure — keep existing count, do not crash
        }
      };

      useEffect(() => {
        // Fetch immediately on mount
        fetchCount();

        // Then poll every 30 seconds
        intervalRef.current = setInterval(fetchCount, POLL_INTERVAL_MS);

        return () => {
          if (intervalRef.current) {
            clearInterval(intervalRef.current);
          }
        };
      }, []);

      return count;
    }
    ```

    Note: apiFetch handles 401 auto-refresh (from 02-04 pattern). No additional auth handling needed here.
