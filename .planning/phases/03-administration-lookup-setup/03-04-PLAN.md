```typescript
export interface AuditLogEntry {
  id: number;
  user: string;          // username or "System"
  action: "CREATE" | "UPDATE" | "DELETE" | "LOGIN" | "LOGOUT" | "IMPERSONATE";
  model_name: string;    // e.g. "User", "Customer", "Port"
  object_id: string;

  changes: Record<string, unknown>;
  ip_address: string | null;
  timestamp: string;     // ISO datetime
}

export interface AuditLogFilters {
  user?: string;
  action?: string;
  module?: string;
  date_from?: string;
  date_to?: string;
  search?: string;
  page?: number;
}
```

- Date range: two date inputs (From / To)
- Search: text input for object_repr search
- "Apply Filters" button and "Clear" button
- Calls parent onFilterChange(filters: AuditLogFilters) callback
- Debounce search input (300ms)
