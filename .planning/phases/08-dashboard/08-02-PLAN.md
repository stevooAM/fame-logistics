**frontend/src/hooks/use-dashboard.ts** — Polling hook modelled exactly on useApprovalBadge:
```typescript
"use client";

import { useEffect, useState, useRef, useCallback } from "react";
import { apiFetch } from "@/lib/api";
import type { DashboardResponse, FeedPage, ActivityEntry } from "@/types/dashboard";

const POLL_INTERVAL_MS = 30_000;

export interface UseDashboardResult {
  kpis: DashboardResponse["kpis"] | null;
  feed: ActivityEntry[];
  feedTotal: number;
  feedNext: string | null;
  loading: boolean;
  loadingMore: boolean;
  loadMore: () => Promise<void>;
}

export function useDashboard(): UseDashboardResult {
  const [data, setData] = useState<DashboardResponse | null>(null);
  const [extraEntries, setExtraEntries] = useState<ActivityEntry[]>([]);
  const [feedNext, setFeedNext] = useState<string | null>(null);
  const [feedTotal, setFeedTotal] = useState(0);
  const [loadingMore, setLoadingMore] = useState(false);
  const intervalRef = useRef<ReturnType<typeof setInterval> | null>(null);

  const fetchDashboard = useCallback(async () => {
    try {
      const result = await apiFetch<DashboardResponse>("/api/dashboard/");
      setData(result);
      setFeedNext(result.feed.next);
      setFeedTotal(result.feed.count);
      // Reset extra entries on each poll refresh (feed base resets to latest 10)
      setExtraEntries([]);
    } catch {
      // Non-blocking — keep existing data on error (no flash to zero)
    }
  }, []);

  useEffect(() => {
    fetchDashboard();
    intervalRef.current = setInterval(fetchDashboard, POLL_INTERVAL_MS);
    return () => {
      if (intervalRef.current !== null) clearInterval(intervalRef.current);
    };
  }, [fetchDashboard]);

  const loadMore = useCallback(async () => {
    if (!feedNext) return;
    setLoadingMore(true);
    try {
      const page = await apiFetch<FeedPage>(feedNext);
      setExtraEntries(prev => [...prev, ...page.results]);
      setFeedNext(page.next);
    } catch {
      // Non-blocking
    } finally {
      setLoadingMore(false);
    }
  }, [feedNext]);

  const allEntries = [
    ...(data?.feed.results ?? []),
    ...extraEntries,
  ];

  return {
    kpis: data?.kpis ?? null,
    feed: allEntries,
    feedTotal,
    feedNext,
    loading: data === null,
    loadingMore,
    loadMore,
  };
}
```

  function relativeTime(isoTimestamp: string): string {
    const diffMs = Date.now() - new Date(isoTimestamp).getTime();
    const diffSec = Math.round(diffMs / 1000);
    const rtf = new Intl.RelativeTimeFormat("en", { numeric: "auto" });
    if (Math.abs(diffSec) < 60) return rtf.format(-diffSec, "second");
    const diffMin = Math.round(diffSec / 60);
    if (Math.abs(diffMin) < 60) return rtf.format(-diffMin, "minute");
    const diffHr = Math.round(diffMin / 60);
    if (Math.abs(diffHr) < 24) return rtf.format(-diffHr, "hour");
    const diffDay = Math.round(diffHr / 24);
    return rtf.format(-diffDay, "day");
  }
